#include <Arduino.h>
#include <math.h>

// =====================================================
// BoxDef 必须最先定义，Arduino IDE 不自动前向声明结构体
// =====================================================
struct BoxDef {
  int    id;
  double angleMin;
  double angleMax;
};

// ---------- 行B：boxes 11~16，中间玻璃区 0°~+24° ----------
const BoxDef rowB[] = {
  {11, -40.0, -24.0},
  {12, -24.0, -16.0},
  {13, -16.0,  -8.0},
  {14,  -8.0,   0.0},
  {15,  24.0,  32.0},
  {16,  32.0,  40.0},
};
const int rowB_count = 6;

// ---------- 行C：boxes 17~22，布局同行B ----------
const BoxDef rowC[] = {
  {17, -40.0, -24.0},
  {18, -24.0, -16.0},
  {19, -16.0,  -8.0},
  {20,  -8.0,   0.0},
  {21,  24.0,  32.0},
  {22,  32.0,  40.0},
};
const int rowC_count = 6;

// ---------- 行D：boxes 23~27，右侧 0°~+40° ----------
const BoxDef rowD_top[] = {
  {23,   0.0,  16.0},
  {24,  16.0,  32.0},
  {25,  32.0,  40.0},
};
const BoxDef rowD_bot[] = {
  {26,   0.0,  16.0},
  {27,  16.0,  32.0},
};
const int rowD_top_count = 3;
const int rowD_bot_count = 2;

/**
 * LD2450 图1正面箱体精确映射系统 v2
 *
 * 图1 箱体布局（雷达正前方视角，左负右正）：
 *
 *  行A 远距离  (1.40~2.00m)  boxes  1~10  均匀10格，每格8°，-40°~+40°
 *  行B 中远距离(0.90~1.40m)  boxes 11~16  左4格+右2格，中间玻璃不触发
 *  行C 中近距离(0.50~0.90m)  boxes 17~22  左4格+右2格，中间玻璃不触发
 *  行D 近距离  (0.25~0.50m)  boxes 23~27  仅右侧0°~+40°
 *  行E 极近距离(0~0.25m)     boxes 28~38  均匀11格，每格≈7.27°，-40°~+40°
 *
 * ESP32  RX=GPIO4  TX=GPIO5
 */

// =====================================================
// 硬件配置
// =====================================================
#define RADAR_RX_PIN  4
#define RADAR_TX_PIN  5
#define RADAR_BAUD    256000
#define FRAME_LEN     30
#define BUFFER_SIZE   128

// =====================================================
// 调试开关
// =====================================================
#define PRINT_EACH_BYTE false

// =====================================================
// 距离层分界（米）
// =====================================================
#define DIST_E_MAX   0.25
#define DIST_D_MAX   0.50
#define DIST_C_MAX   0.90
#define DIST_B_MAX   1.40
#define DIST_A_MAX   2.00

// =====================================================
// 交叉融合半宽（度）
// =====================================================
#define BLEND_HALF   2.5

// =====================================================
// 环形缓冲区
// =====================================================
uint8_t ringBuf[BUFFER_SIZE];
int ringHead  = 0;
int ringTail  = 0;
int ringCount = 0;

// =====================================================
// 目标结构
// =====================================================
struct Target {
  double x, y;
  double distance;
  double angle;
  double speed;
  double resolution;
  bool   valid;
};
Target targets[3];

// =====================================================
// 系统统计
// =====================================================
unsigned long lastTargetPrint = 0;
unsigned long lastStatusPrint = 0;
uint32_t byteCount       = 0;
uint32_t validFrameCount = 0;
uint32_t fakeHeadCount   = 0;
uint32_t noHeadCount     = 0;

// =====================================================
// 映射结果
// =====================================================
int frontMainBox  = -1;
int frontBlendBox = -1;

// =====================================================
// LD2450 解码辅助
// =====================================================
double calcCoordinate(int low, int high) {
  int raw = low + high * 256;
  if (raw & 0x8000) return (double)raw - 32768.0;
  return (double)(0 - raw);
}
double calcUint16(int low, int high) {
  return (double)(low + high * 256);
}

// =====================================================
// 通用：在 BoxDef 数组中按角度查找箱体 ID
// =====================================================
int findBox(const BoxDef* defs, int count, double angle) {
  for (int i = 0; i < count; i++) {
    double lo = defs[i].angleMin;
    double hi = defs[i].angleMax;
    bool inRange = (i == count - 1)
      ? (angle >= lo && angle <= hi)
      : (angle >= lo && angle <  hi);
    if (inRange) return defs[i].id;
  }
  return -1;
}

// =====================================================
// 融合：在同一行找边界相邻箱
// =====================================================
int blendInRow(const BoxDef* defs, int count, double angle, int mainId) {
  for (int i = 0; i < count; i++) {
    if (defs[i].id != mainId) continue;
    if (angle < defs[i].angleMin + BLEND_HALF && i > 0)
      return defs[i - 1].id;
    if (angle > defs[i].angleMax - BLEND_HALF && i < count - 1)
      return defs[i + 1].id;
    break;
  }
  return -1;
}

// =====================================================
// 行A 箱体（1~10），每格8°
// =====================================================
int rowA_main(double angle) {
  int col = (int)floor((angle + 40.0) / 8.0) + 1;
  if (col < 1)  col = 1;
  if (col > 10) col = 10;
  return col;
}

// =====================================================
// 行E 箱体（28~38），每格 80/11°
// =====================================================
int rowE_main(double angle) {
  const double step = 80.0 / 11.0;
  int col = (int)floor((angle + 40.0) / step);
  if (col < 0)  col = 0;
  if (col > 10) col = 10;
  return 28 + col;
}

// =====================================================
// 主映射函数
// =====================================================
void mapFrontBox(double angle, double distance) {
  frontMainBox  = -1;
  frontBlendBox = -1;

  if (angle < -40.0 || angle > 40.0) return;
  if (distance <= 0 || distance > DIST_A_MAX) return;

  // 行A
  if (distance > DIST_B_MAX) {
    frontMainBox = rowA_main(angle);
    double localPos = fmod((angle + 40.0), 8.0);
    if (localPos < BLEND_HALF && frontMainBox > 1)
      frontBlendBox = frontMainBox - 1;
    else if (localPos > (8.0 - BLEND_HALF) && frontMainBox < 10)
      frontBlendBox = frontMainBox + 1;
    return;
  }

  // 行B
  if (distance > DIST_C_MAX) {
    frontMainBox = findBox(rowB, rowB_count, angle);
    if (frontMainBox == -1) return;
    frontBlendBox = blendInRow(rowB, rowB_count, angle, frontMainBox);
    return;
  }

  // 行C
  if (distance > DIST_D_MAX) {
    frontMainBox = findBox(rowC, rowC_count, angle);
    if (frontMainBox == -1) return;
    frontBlendBox = blendInRow(rowC, rowC_count, angle, frontMainBox);
    return;
  }

  // 行D
  if (distance > DIST_E_MAX) {
    if (angle < 0.0) return;  // 左半无箱体
    frontMainBox = findBox(rowD_top, rowD_top_count, angle);
    if (frontMainBox == -1) return;
    frontBlendBox = findBox(rowD_bot, rowD_bot_count, angle);
    if (frontBlendBox == frontMainBox) frontBlendBox = -1;
    return;
  }

  // 行E
  frontMainBox = rowE_main(angle);
  const double step = 80.0 / 11.0;
  double localPos = fmod((angle + 40.0), step);
  if (localPos < BLEND_HALF && frontMainBox > 28)
    frontBlendBox = frontMainBox - 1;
  else if (localPos > (step - BLEND_HALF) && frontMainBox < 38)
    frontBlendBox = frontMainBox + 1;
}

// =====================================================
// setup
// =====================================================
void setup() {
  Serial.begin(115200);
  delay(500);
  Serial1.begin(RADAR_BAUD, SERIAL_8N1, RADAR_RX_PIN, RADAR_TX_PIN);
  Serial.println();
  Serial.println("======================================");
  Serial.println("LD2450 图1正面箱体精确映射 v2");
  Serial.println("======================================");
  while (Serial1.available()) Serial1.read();
}

// =====================================================
// loop
// =====================================================
void loop() {
  while (Serial1.available()) {
    uint8_t b = Serial1.read();
    byteCount++;
    ringBuf[ringHead] = b;
    ringHead = (ringHead + 1) % BUFFER_SIZE;
    if (ringCount < BUFFER_SIZE) {
      ringCount++;
    } else {
      ringTail = (ringTail + 1) % BUFFER_SIZE;
    }
    if (PRINT_EACH_BYTE) {
      if (b < 0x10) Serial.print("0");
      Serial.print(b, HEX);
      Serial.print(" ");
    }
  }

  parseFromRingBuffer();

  if (millis() - lastStatusPrint >= 2000) {
    Serial.print("[STAT] bytes=");   Serial.print(byteCount);
    Serial.print(" validFrames=");  Serial.print(validFrameCount);
    Serial.print(" fakeHead=");     Serial.print(fakeHeadCount);
    Serial.print(" noHead=");       Serial.print(noHeadCount);
    Serial.print(" ringCount=");    Serial.println(ringCount);
    lastStatusPrint = millis();
  }
}

// =====================================================
// 解析环形缓冲区
// =====================================================
void parseFromRingBuffer() {
  while (ringCount >= FRAME_LEN) {
    int frameStartOffset = -1;
    for (int i = 0; i <= ringCount - FRAME_LEN; i++) {
      int p0 = (ringTail + i + 0) % BUFFER_SIZE;
      int p1 = (ringTail + i + 1) % BUFFER_SIZE;
      int p2 = (ringTail + i + 2) % BUFFER_SIZE;
      int p3 = (ringTail + i + 3) % BUFFER_SIZE;
      if (ringBuf[p0]==0xAA && ringBuf[p1]==0xFF &&
          ringBuf[p2]==0x03 && ringBuf[p3]==0x00) {
        frameStartOffset = i;
        break;
      }
    }
    if (frameStartOffset == -1) {
      int drop = ringCount - FRAME_LEN + 1;
      ringTail  = (ringTail + drop) % BUFFER_SIZE;
      ringCount -= drop;
      noHeadCount++;
      return;
    }
    if (frameStartOffset > 0) {
      ringTail  = (ringTail + frameStartOffset) % BUFFER_SIZE;
      ringCount -= frameStartOffset;
    }
    if (ringCount < FRAME_LEN) return;

    int tail1 = (ringTail + 28) % BUFFER_SIZE;
    int tail2 = (ringTail + 29) % BUFFER_SIZE;
    if (ringBuf[tail1]==0x55 && ringBuf[tail2]==0xCC) {
      uint8_t frame[FRAME_LEN];
      for (int i = 0; i < FRAME_LEN; i++)
        frame[i] = ringBuf[(ringTail + i) % BUFFER_SIZE];
      validFrameCount++;
      parseFrame(frame);
      if (millis() - lastTargetPrint >= 300) {
        printTargets();
        lastTargetPrint = millis();
      }
      ringTail  = (ringTail + FRAME_LEN) % BUFFER_SIZE;
      ringCount -= FRAME_LEN;
    } else {
      ringTail  = (ringTail + 1) % BUFFER_SIZE;
      ringCount--;
      fakeHeadCount++;
    }
  }
}

// =====================================================
// 解析一帧
// =====================================================
void parseFrame(uint8_t *frame) {
  int off = 4;
  for (int i = 0; i < 3; i++) {
    int xL=frame[off+0], xH=frame[off+1];
    int yL=frame[off+2], yH=frame[off+3];
    int sL=frame[off+4], sH=frame[off+5];
    int rL=frame[off+6], rH=frame[off+7];
    off += 8;
    bool allZero = !xL&&!xH&&!yL&&!yH&&!sL&&!sH&&!rL&&!rH;
    if (allZero) { targets[i].valid = false; continue; }
    targets[i].x          = calcCoordinate(xL, xH);
    targets[i].y          = calcCoordinate(yL, yH);
    targets[i].speed      = calcCoordinate(sL, sH);
    targets[i].resolution = calcUint16(rL, rH);
    targets[i].distance   = sqrt(targets[i].x*targets[i].x +
                                 targets[i].y*targets[i].y) / 1000.0;
    targets[i].angle      = atan2(targets[i].x, targets[i].y) * 180.0 / PI;
    targets[i].valid      = true;
  }
}

// =====================================================
// 输出结果
// =====================================================
void printTargets() {
  Serial.println();
  Serial.println("======================================");
  Serial.println("     LD2450 图1正面箱体监视");
  Serial.println("======================================");

  bool anyValid = false;
  for (int i = 0; i < 3; i++) {
    if (!targets[i].valid) continue;
    if (targets[i].distance > DIST_A_MAX) continue;
    if (targets[i].angle < -40.0 || targets[i].angle > 40.0) continue;

    anyValid = true;
    mapFrontBox(targets[i].angle, targets[i].distance);

    String hZone;
    if      (targets[i].angle < -20) hZone = "左侧";
    else if (targets[i].angle <  -5) hZone = "左中";
    else if (targets[i].angle <=  5) hZone = "中央";
    else if (targets[i].angle <= 20) hZone = "右中";
    else                             hZone = "右侧";

    String dZone;
    if      (targets[i].distance <= DIST_E_MAX) dZone = "极近(行E)";
    else if (targets[i].distance <= DIST_D_MAX) dZone = "近距(行D)";
    else if (targets[i].distance <= DIST_C_MAX) dZone = "中近(行C)";
    else if (targets[i].distance <= DIST_B_MAX) dZone = "中远(行B)";
    else                                         dZone = "远距(行A)";

    Serial.println();
    Serial.print("目标 "); Serial.println(i + 1);
    Serial.println("--------------------------------------");
    Serial.print("角度: "); Serial.print(targets[i].angle, 1); Serial.println(" deg");
    Serial.print("距离: "); Serial.print(targets[i].distance, 3); Serial.println(" m");
    Serial.print("速度: "); Serial.print(targets[i].speed, 0); Serial.println(" cm/s");
    Serial.print("位置: "); Serial.print(hZone); Serial.print(" / "); Serial.println(dZone);
    Serial.println();
    Serial.println("[图1 正面箱体]");
    if (frontMainBox == -1) {
      Serial.println("  → 玻璃/无箱体区域，不触发");
    } else {
      Serial.print("  主触发: "); Serial.println(frontMainBox);
      if (frontBlendBox != -1) {
        Serial.print("  融合:   "); Serial.println(frontBlendBox);
      }
      Serial.print("  → 人在 "); Serial.print(hZone);
      Serial.print(" "); Serial.print(dZone);
      Serial.print(" 触发箱 "); Serial.print(frontMainBox);
      if (frontBlendBox != -1) { Serial.print("+"); Serial.print(frontBlendBox); }
      Serial.println();
    }
    Serial.println("--------------------------------------");
  }

  if (!anyValid) {
    Serial.println("当前无有效目标（超出范围）");
  }
  Serial.println("======================================");
}
